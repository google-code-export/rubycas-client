    def parse_successful(elm)
      super(elm)
#      puts "proxy_successful"
      proxies = elm.elements["cas:proxies"]
      if proxies
        proxies.elements.each("cas:proxy") { |prox|
          @proxy_list ||= []
          @proxy_list << prox.text.strip
        }
      end
    end
