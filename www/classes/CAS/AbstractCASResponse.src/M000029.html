    def parse(str)
#      puts "parsing... #{str}"
      doc = REXML::Document.new str
      resp = doc.elements["cas:serviceResponse"].elements[1]
#      puts "resp... #{resp.name}"
      if successful_response? resp
        parse_successful(resp)
      else
        parse_unsuccessful(resp)
      end
    end
